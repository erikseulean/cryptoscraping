def color(val):
    if val < 0:
        color = 'red'
    elif val == 0:
        color = 'yellow'
    elif val > 0:
        color = 'green'
    return "background-color: {}; color: white".format(color)

def color(val):
    bg = 'white'
    color = 'black'
    if val < 0:
        color = 'white'
        bg = 'red'
    elif val == 0:
        color = 'gray'
        bg = 'white'
    elif val > 0:
        color = 'white'
        bg = 'green'
    return "background-color: {}; color: {}; text-align: right".format(bg, color)
